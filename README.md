    │   ├── index.html

    ├── src

    │   │   ├── App.test.tsx
    │   │   └── App.tsx
    │   ├── assets
    │   │   ├── icons
    │   │   │   ├── index.ts
    │   │   │   └── svg
    │   │   │       └── logo.svg
    │   │   ├── images

    ├── styleguide.scss
    ├── tsconfig.json


## setup
```sh
git clone https://github.com/wmadfaa/reactjs-boilerplate--typescript.git <project name>

cd <project name>

yarn install
```

## start
```sh
yarn start
```

## build
```sh
yarn build
```

## start style guide
```sh
yarn sg
```

## build style guide
```sh
yarn sg:build
```
